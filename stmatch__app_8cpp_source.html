19 void STMATCHApp::run() {
20  UTIL::TimePoint start_time = std::chrono::steady_clock::now();
21  STMATCH mm_model(network_, ng_);
22  const STMATCHConfig &stmatch_config =
23  config_.stmatch_config;
24  IO::GPSReader reader(config_.gps_config);
25  IO::CSVMatchResultWriter writer(config_.result_config.file,
26  config_.result_config.output_config);
27  // Start map matching
28  int progress = 0;
29  int points_matched = 0;
30  int total_points = 0;
31  int step_size = 100;
32  if (config_.step > 0) step_size = config_.step;
33  SPDLOG_INFO("Progress report step {}", step_size);
34  UTIL::TimePoint corrected_begin = std::chrono::steady_clock::now();
35  SPDLOG_INFO("Start to match trajectories");
36  if (config_.use_omp){
37  SPDLOG_INFO("Run map matching parallelly");
38  int buffer_trajectories_size = 100000;
39  while (reader.has_next_trajectory()) {
40  std::vector<Trajectory> trajectories =
41  reader.read_next_N_trajectories(buffer_trajectories_size);
42  int trajectories_fetched = trajectories.size();
43  #pragma omp parallel for
44  for (int i = 0; i < trajectories_fetched; ++i) {
45  Trajectory &trajectory = trajectories[i];
46  int points_in_tr = trajectory.geom.get_num_points();
47  MM::MatchResult result = mm_model.match_traj(
48  trajectory, stmatch_config);
49  writer.write_result(result);
50  #pragma omp critical
51  if (!result.cpath.empty()) {
52  points_matched += points_in_tr;
53  }
54  total_points += points_in_tr;
55  ++progress;
56  if (progress % step_size == 0) {
57  std::stringstream buf;
58  buf << "Progress " << progress << '\n';
59  std::cout << buf.rdbuf();
60  }
61  }
62  }
63  } else {
64  SPDLOG_INFO("Run map matching in single thread");
65  while (reader.has_next_trajectory()) {
66  if (progress % step_size == 0) {
67  SPDLOG_INFO("Progress {}", progress);
68  }
69  Trajectory trajectory = reader.read_next_trajectory();
70  int points_in_tr = trajectory.geom.get_num_points();
71  MM::MatchResult result = mm_model.match_traj(
72  trajectory, stmatch_config);
73  writer.write_result(result);
74  if (!result.cpath.empty()) {
75  points_matched += points_in_tr;
76  }
77  total_points += points_in_tr;
78  ++progress;
79  }
80  }
81  SPDLOG_INFO("MM process finished");
82  UTIL::TimePoint end_time = std::chrono::steady_clock::now();
83  double time_spent = std::chrono::duration_cast<std::chrono::milliseconds>(
84  end_time - start_time).count() / 1000.;
85  double time_spent_exclude_input = std::chrono::duration_cast<
86  std::chrono::milliseconds>(end_time - corrected_begin).count() / 1000.;
87  SPDLOG_INFO("Time takes {}", time_spent);
88  SPDLOG_INFO("Time takes excluding input {}", time_spent_exclude_input);
89  SPDLOG_INFO("Finish map match total points {} matched {}",
90  total_points, points_matched);
91  SPDLOG_INFO("Matched percentage: {}", points_matched / (double) total_points);
92  SPDLOG_INFO("Point match speed: {}", points_matched / time_spent);
93  SPDLOG_INFO("Point match speed (excluding input): {}",
94  points_matched / time_spent_exclude_input);
95  SPDLOG_INFO("Time takes {}", time_spent);
96 };
