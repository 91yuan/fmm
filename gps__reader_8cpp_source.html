61  if (timestamp_idx < 0) {

63  }
64  if (wkbFlatten(ogrFDefn->GetGeomType()) != wkbLineString) {
65  SPDLOG_CRITICAL("Geometry type is {}, which should be linestring",
66  OGRGeometryTypeToName(ogrFDefn->GetGeomType()));
67  GDALClose(poDS);
68  std::exit(EXIT_FAILURE);
69  } else {
70  SPDLOG_DEBUG("Geometry type is {}",
71  OGRGeometryTypeToName(ogrFDefn->GetGeomType()));
72  }

243 Trajectory CSVPointReader::read_next_trajectory() {
244  // Read the geom idx column into a trajectory
245  std::string intermediate;
246  FMM::CORE::LineString geom;
247  std::vector<double> timestamps;
248  bool on_same_trajectory = true;
249  bool first_observation = true;
250  int trid = -1;
251  int prev_id = -1;
252  double prev_timestamp = -1.0;
253  std::string line;
254  while (on_same_trajectory && has_next_trajectory()) {
255  if (prev_line.empty()) {
256  std::getline(ifs, line);
257  } else {
258  line = prev_line;
259  prev_line.clear();
260  }
261  std::stringstream ss(line);
262  int id = 0;
263  double x = 0, y = 0;
264  double timestamp = 0;
265  int index = 0;
266  while (std::getline(ss, intermediate, delim)) {
267  if (index == id_idx) {
268  id = std::stoi(intermediate);
269  }
270  if (index == x_idx) {
271  x = std::stof(intermediate);
272  }
273  if (index == y_idx) {
274  y = std::stof(intermediate);
275  }
276  if (index == timestamp_idx) {
277  timestamp = std::stof(intermediate);
278  }
279  ++index;
280  }
281  if (prev_id == id || first_observation) {
282  geom.add_point(x, y);
283  if (has_timestamp())
284  timestamps.push_back(timestamp);
285  }
286  if (prev_id != id && !first_observation) {
287  on_same_trajectory = false;
288  trid = prev_id;
289  }
290  first_observation = false;
291  prev_id = id;
292  if (!on_same_trajectory) {
293  prev_line = line;
294  }
295  }
296  if (!has_next_trajectory()){
297  trid = prev_id;
298  }
299  return Trajectory{trid, geom, timestamps};
300 }
