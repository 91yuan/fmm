22  int progress = 0;

30  if (config_.use_omp){
31  SPDLOG_INFO("Run map matching parallelly");

41  MM::MatchResult result = mm_model.match_traj(
42  trajectory, fmm_config);
43  writer.write_result(result);
44  #pragma omp critical
45  if (!result.cpath.empty()) {
46  points_matched += points_in_tr;
47  }
48  total_points += points_in_tr;
49  ++progress;
50  if (progress % step_size == 0) {
51  std::stringstream buf;
52  buf << "Progress " << progress << '\n';
53  std::cout << buf.rdbuf();
54  }
55  }
56  }
57  } else {
58  SPDLOG_INFO("Run map matching in single thread");
59  while (reader.has_next_trajectory()) {
60  if (progress % step_size == 0) {
61  SPDLOG_INFO("Progress {}", progress);
62  }
63  Trajectory trajectory = reader.read_next_trajectory();
64  int points_in_tr = trajectory.geom.get_num_points();
65  MM::MatchResult result = mm_model.match_traj(
66  trajectory, fmm_config);
67  writer.write_result(result);
68  if (!result.cpath.empty()) {
69  points_matched += points_in_tr;
70  }
71  total_points += points_in_tr;
72  ++progress;
73  }
74  }
75  SPDLOG_INFO("MM process finished");
76  UTIL::TimePoint end_time = std::chrono::steady_clock::now();
77  double time_spent = std::chrono::duration_cast<std::chrono::milliseconds>(
78  end_time - start_time).count() / 1000.;
79  double time_spent_exclude_input = std::chrono::duration_cast<
80  std::chrono::milliseconds>(end_time - corrected_begin).count() / 1000.;
81  SPDLOG_INFO("Time takes {}", time_spent);
82  SPDLOG_INFO("Time takes excluding input {}", time_spent_exclude_input);
83  SPDLOG_INFO("Finish map match total points {} matched {}",
84  total_points, points_matched);
85  SPDLOG_INFO("Matched percentage: {}", points_matched / (double) total_points);
86  SPDLOG_INFO("Point match speed: {}", points_matched / time_spent);
87  SPDLOG_INFO("Point match speed (excluding input): {}",

89  SPDLOG_INFO("Time takes {}", time_spent);
